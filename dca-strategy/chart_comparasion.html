import numpy as np
import pandas as pd
import talib as ta
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
from R_DCA_T2 import load_model, preprocess_data, create_labels
from generate_data_for_charts import calculate_valuation_by_profit, calculate_valuation_by_mrvr, calculate_valuation_by_oscillator, calculate_valuation_by_ALL_IN, calculate_valuation_by_DCA, calculate_valuation_by_RDCA
import warnings
warnings.filterwarnings("ignore", category=pd.errors.SettingWithCopyWarning)

# Load m hnh  lu
model = load_model("R_DCA_T2_trained.keras")

from matplotlib.ticker import FuncFormatter

# Hm formatter  hin th gi tr Bitcoin ngn gn
def y_format(value, tick_number):
    return f'{int(value/1000)}k'  # Chuyn gi tr thnh nh dng '68k'
# Load v sort li data theo ngy tng dn  render chart
data = pd.read_csv("../../data/merged_data_train_test_output.csv").sort_values(by='Date', ascending=True)
# data = create_labels(data)
_, X_test_scaled, _, _, _, _, test_data = preprocess_data(data)

# D on kt qu
predictions = model.predict(X_test_scaled)
# predicted_target = np.argmax(predictions[0], axis=1)

# p dng ngng thay v dng argmax
predicted_target = []
threshold = 0.1  # V d: ngng ang mun l 10%
for i, pred in enumerate(predictions[0]):
    target = 0
    if pred[1] > threshold:  # Lp Buy c xc sut > ngng
        target = 1
    elif pred[2] > threshold:  # Lp Sell c xc sut > ngng
        target = 2
    else:
        target = 0
    predicted_target.append(target)  # Hold nu khng c lp no vt ngng
    # print(f"Row {i} - Hold: {pred[0]:.4f}, Buy: {pred[1]:.4f}, Sell: {pred[2]:.4f} Target {target}")

# nh ngha cc mng tng ng vi cc gi tr Target v Confidence Level

# Mappings cho Target
target_mapping = {"Hold": 0, "Buy": 1, "Sell": 2}

# Gi tr khi to ngy u tin trong chu k

portfolio_percentage = 0.3

# Render RDCA chart
def render_all_charts(from_date_str, to_date_str, days_interval = 7, rdca_buy_sell_btc = 1, initial_cash=1000000, btc_initial = 5, daily_investment = 20000):
    from_date = pd.to_datetime(from_date_str)  # Chuyn i thnh Timestamp
    to_date = pd.to_datetime(to_date_str)    # Chuyn i thnh Timestamp

    # Chuyn i ct 'Date' trong test_data thnh kiu datetime
    test_data['Date'] = pd.to_datetime(test_data['Date'])
    # print(f"------- test_data['Date'] {test_data['Date']}")

    # Lc d liu theo khong thi gian from_date n to_date
    filtered_data = test_data[(test_data['Date'] >= from_date) & (test_data['Date'] <= to_date)]

    # Chun b d liu  v biu 
    dates = filtered_data['Date'].dt.strftime('%Y-%m-%d').values  # Ngy trong khong thi gian t from_date n to_date, nh dng chui 'YYYY-MM-DD'
    prices = filtered_data['Price'].values  # Gi Bitcoin trong khong thi gian 

    # To mng kt qu theo ngy, target v confidence level
    result_array = []

    for i in range(len(test_data['Date'])):
        date = test_data['Date'].iloc[i]  # Ly ngy t d liu test
        target = target_labels[predicted_target[i]]
        # Ch thm kt qu nu ngy nm trong khong from_date n to_date
        if from_date <= date <= to_date:
            # result_array.append([date.strftime('%Y-%m-%d'), target, price])  # Chuyn i 'date' sang nh dng chui 'YYYY-MM-DD'
            result_array.append([date.strftime('%Y-%m-%d'), target])
    
    # # Danh sch ngy, target, gi Bitcoin t d liu
    dates = pd.to_datetime([row[0] for row in result_array])  # Convert list to pandas datetime
    # print(f"------- dates {dates}")

    # Tnh ton Valuation cho phng php all in BTC
    ratio_profit_data = filtered_data['Ratio_Profit'].values
    ratio_mvrv_data = filtered_data['Ratio_MRRV'].values
    ratio_osci_data = filtered_data['Oscillator'].values
    # Cp nht valuation theo target v gi Bitcoin
    targets = [target_mapping[row[1]] for row in result_array]
    
    valuation_by_profit, btc_holdings_by_profit = calculate_valuation_by_profit(dates, prices, ratio_profit_data, initial_cash, rdca_buy_sell_btc, btc_initial)
    valuation_by_mvrv, btc_holdings_by_mvrv = calculate_valuation_by_mrvr(dates, prices, ratio_mvrv_data, initial_cash, rdca_buy_sell_btc, btc_initial)
    valuation_by_osci, btc_holdings_by_osci = calculate_valuation_by_oscillator(dates, prices, ratio_osci_data, initial_cash, rdca_buy_sell_btc, btc_initial)
    valuation_by_ALL_IN, btc_holdings_by_ALL_IN = calculate_valuation_by_ALL_IN(dates, prices, initial_cash, btc_initial = 5)
    valuation_by_DCA, btc_holdings_by_DCA = calculate_valuation_by_DCA(dates, prices, initial_cash, btc_initial, daily_investment)
    valuation_by_RDCA, btc_holdings_by_RDCA = calculate_valuation_by_RDCA(targets, dates, prices, initial_cash, btc_initial, rdca_buy_sell_btc)
    
    # V biu  vi valuation
    fig, ax1 = plt.subplots(figsize=(12, 6))

    ax1.set_ylabel("BTC")
    ax1.set_xticks(dates)
    ax1.set_xticklabels(dates.strftime('%Y-%m-%d'), rotation=1)
    ax1.xaxis.set_major_locator(mdates.DayLocator(interval=days_interval))  # Hin th interval ngy
    ax1.scatter(dates, prices, label="BTC Price", color='black', marker='.')
    ax1.legend(loc="upper right")

    # V biu  valuation (gi tr ti sn)
    ax3 = ax1.twinx()
    ax3.set_ylabel("Valuation")

    dates = pd.to_datetime(dates)
    # V biu  valuation (DCA - ng mu tm)
    ax3.plot(dates, valuation_by_profit, label=f"By Profit", color='fuchsia', linestyle='-', marker='')
    ax3.plot(dates, valuation_by_mvrv, label=f"By MVRV", color='brown', linestyle='-', marker='')
    ax3.plot(dates, valuation_by_osci, label=f"By OSCI", color='darkblue', linestyle='-', marker='')
    ax3.plot(dates, valuation_by_ALL_IN, label=f"ALL IN", color='orange', linestyle='-', marker='')
    ax3.plot(dates, valuation_by_DCA, label=f"DCA", color='green', linestyle='-', marker='')
    ax3.plot(dates, valuation_by_RDCA, label=f"RDCA", color='red', linestyle='-', marker='')

    ax1.yaxis.set_major_formatter(FuncFormatter(y_format))
    ax3.legend(loc="upper left")

    # Thm tiu  v hin th biu 
    plt.title(f"From {from_date_str} to {to_date_str} Investment: {initial_cash / 1000:,.0f}k USD in Cash & {btc_initial} BTC")
    plt.tight_layout()
    plt.show()

    print("--------- RDCA ----------")
    print(f"[RDCA] Valuation n ngy {to_date_str}: {valuation_by_RDCA[-1] / 1000:,.0f}k USD (Cash + {btc_holdings_by_RDCA:.2f} BTC)")
    print("--------- ALL IN ----------")
    print(f"[ALL IN] Valuation n ngy {to_date_str}: {valuation_by_ALL_IN[-1] / 1000:,.0f}k USD ~= {btc_holdings_by_ALL_IN:.2f} BTC")
    print("--------- Traditional DCA ----------")
    print(f"[DCA] Valuation n ngy {to_date_str}: {valuation_by_DCA[-1] / 1000:,.0f}k USD ~= {btc_holdings_by_DCA:.2f} BTC")
    print("--------- PROFIT ----------")
    print(f"[By Profit] Valuation Profit n ngy {to_date_str}: {valuation_by_profit[-1] / 1000:,.0f}k USD (Cash + {btc_holdings_by_profit:.2f} BTC)")
    print("--------- MVRV ----------")
    print(f"[By MRVR] Valuation MVRV n ngy {to_date_str}: {valuation_by_mvrv[-1] / 1000:,.0f}k USD (Cash + {btc_holdings_by_mvrv:.2f} BTC)")
    print("--------- OSCI ----------")
    print(f"[By Oscillator] Valuation Oscillator n ngy {to_date_str}: {valuation_by_osci[-1] / 1000:,.0f}k (Cash + {btc_holdings_by_osci:.2f} BTC)")
