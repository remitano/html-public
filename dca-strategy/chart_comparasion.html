        date = test_data['Date'].iloc[i]  # Lấy ngày từ dữ liệu test
        target = target_labels[predicted_target[i]]
        # Chỉ thêm kết quả nếu ngày nằm trong khoảng from_date đến to_date
        if from_date <= date <= to_date:
            # result_array.append([date.strftime('%Y-%m-%d'), target, price])  # Chuyển đổi 'date' sang định dạng chuỗi 'YYYY-MM-DD'
            result_array.append([date.strftime('%Y-%m-%d'), target])

        current_btc_price = prices[0]
        current_valuation = initial_cash
        remaining_cash = initial_cash
        btc_holdings = btc_initial
        result_array = []
        valuation = []
        # print("--------- PROFIT ----------")  
        for i in range(len(dates)):
            # Profit
            bias_amount = 0
            if ratio_profit_data[i] <  0.6 and remaining_cash > 0:  # Buy Bitcoin
                btc_holdings += rdca_buy_sell_btc
                bias_amount = current_btc_price * rdca_buy_sell_btc
            elif ratio_profit_data[i] >  0.9 and remaining_cash > 0 and btc_holdings > 0:  # Sell Bitcoin
                btc_holdings -= rdca_buy_sell_btc
                bias_amount = -current_btc_price * rdca_buy_sell_btc
            # Hold thì mọi thứ vẫn giữ lại không thay đổi về lượng BTC đang giữ
            current_btc_price = prices[i]
            remaining_cash -= bias_amount
            current_valuation = remaining_cash + btc_holdings * prices[i]
            valuation.append(current_valuation)
            result_array.append([ratio_profit_data[i], current_valuation, btc_holdings, remaining_cash])
            # result_array.append(btc_holdings) # Cho mục đích debug
            # result_array.append(remaining_cash)
            # print(f"PROFIT {i} Pro Percentage, Valuation, BTC holding, Cash remain - {result_array[i]}")
        return valuation, btc_holdings
