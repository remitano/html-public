# Refine Chart_R_DCA_vs_DCA
import numpy as np
import pandas as pd
import talib as ta
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
from R_DCA_T2 import load_model, preprocess_data, create_labels
import warnings
warnings.filterwarnings("ignore", category=pd.errors.SettingWithCopyWarning)

# Load mô hình đã lưu
model = load_model("R_DCA_T2_trained.keras")

def load_data_for_chart():
    btc_mvrv = pd.read_csv("../../data/btc-mvrv.csv")
    btc_pi = pd.read_csv("../../data/btc-pi.csv")
    btc_price = pd.read_csv("../../data/btc-price-082024.csv")
    btc_profit_percentage = pd.read_csv("../../data/btc-profit-percentage.csv")
    
    btc_mvrv = btc_mvrv.drop(columns=['API'])
    btc_profit_percentage = btc_profit_percentage.drop(columns=['API'])
    
    # Gộp dữ liệu
    merged_data = btc_mvrv.merge(btc_pi, on='Date').merge(btc_price, on='Date').merge(btc_profit_percentage, on='Date')
    
    # Feature Engineering
    merged_data['RSI'] = ta.RSI(merged_data['Open'], timeperiod=14)
    merged_data['SMA'] = ta.SMA(merged_data['Open'], timeperiod=30)
    merged_data['EMA'] = ta.EMA(merged_data['Open'], timeperiod=30)
    merged_data['Volatility'] = merged_data['Open'].pct_change().rolling(window=14).std()
    merged_data['Bollinger_Upper'], merged_data['Bollinger_Middle'], merged_data['Bollinger_Lower'] = ta.BBANDS(
        merged_data['Open'], timeperiod=20, nbdevup=2, nbdevdn=2, matype=0)

    return merged_data

from matplotlib.ticker import FuncFormatter

# Hàm formatter để hiển thị giá trị Bitcoin ngắn gọn
def y_format(value, tick_number):
    return f'{int(value/1000)}k'  # Chuyển giá trị thành định dạng '68k'
# Load và xử lý dữ liệu
data = load_data_for_chart()
data = create_labels(data)
_, X_test_scaled, _, _, _, _, test_data = preprocess_data(data)

# Dự đoán kết quả
predictions = model.predict(X_test_scaled)
predicted_target = np.argmax(predictions[0], axis=1)


# Định nghĩa các mảng tương ứng với các giá trị Target và Confidence Level
target_labels = {0: "Hold", 1: "Buy", 2: "Sell"}
confidence_labels = {0: "Confidence Low", 1: "Confidence Medium", 2: "Confidence High"}
# Mappings cho Target
target_mapping = {"Hold": 0, "Buy": 1, "Sell": 2}
dca_daily_investment = 20000
# Giá trị khởi tạo ngày đầu tiên trong chu kỳ
# total_cash = 1000000
portfolio_percentage = 0.3
# rdca_buy_sell_btc = 5

# # Định nghĩa khoảng thời gian
# from_date_str = '2024-02-12'

def render_chart_with_param(from_date_str, to_date_str, days_interval = 7, rdca_buy_sell_btc = 5, total_cash=1000000):
    from_date = pd.to_datetime(from_date_str)  # Chuyển đổi thành Timestamp
    to_date = pd.to_datetime(to_date_str)    # Chuyển đổi thành Timestamp

    # Chuyển đổi cột 'Date' trong test_data thành kiểu datetime
    test_data['Date'] = pd.to_datetime(test_data['Date'])

    # Lọc dữ liệu theo khoảng thời gian from_date đến to_date
    filtered_data = test_data[(test_data['Date'] >= from_date) & (test_data['Date'] <= to_date)]

    # Chuẩn bị dữ liệu để vẽ biểu đồ
    dates = filtered_data['Date'].dt.strftime('%Y-%m-%d').values  # Ngày trong khoảng thời gian từ from_date đến to_date, định dạng chuỗi 'YYYY-MM-DD'
    prices = filtered_data['Price'].values  # Giá Bitcoin trong khoảng thời gian đó

    # Tạo mảng kết quả theo ngày, target và confidence level
    result_array = []

    for i in range(len(predicted_target)):
        date = test_data['Date'].iloc[i]  # Lấy ngày từ dữ liệu test
        target = target_labels[predicted_target[i]]  # Chuyển target sang chuỗi Buy/Hold/Sell
        price = test_data['Price'].iloc[i]
        
        # Chỉ thêm kết quả nếu ngày nằm trong khoảng from_date đến to_date
        if from_date <= date <= to_date:
            result_array.append([date.strftime('%Y-%m-%d'), target, price])  # Chuyển đổi 'date' sang định dạng chuỗi 'YYYY-MM-DD'

    # In kết quả result_array để kiểm tra
    # print(f"Filtered result_array from {from_date} to {to_date}:")
    # for row in result_array:
    #     print(row)


    # Danh sách ngày, target, giá Bitcoin từ dữ liệu
    dates = pd.to_datetime([row[0] for row in result_array])  # Convert list to pandas datetime
    targets = [target_mapping[row[1]] for row in result_array]

    current_btc_price = prices[0]

    remaining_cash = total_cash * (1 - portfolio_percentage)
    btc_holdings = (total_cash *  portfolio_percentage) / current_btc_price  # Bắt đầu với 1 triệu USD, đầu tư 30% vào Bitcoin
    valuation = []  # Danh sách chứa giá trị tài sản mỗi ngày (tiền mặt + giá trị Bitcoin)


    # Cập nhật valuation theo target và giá Bitcoin
    for i in range(len(dates)):
        action = targets[i]  # 0 = Hold, 1 = Buy, 2 = Sell
        bias_amount = 0
        if action == 1:  # Buy Bitcoin
            # Tính theo %
            # bias_amount = remaining_cash * portfolio_percentage
            # btc_holdings += bias_amount / current_btc_price
            
            # tính theo số lượng BTC
            btc_holdings += rdca_buy_sell_btc
            bias_amount = current_btc_price
        elif action == 2:  # Sell Bitcoin
            # Tính theo %
            # bias_amount = - prices[i] * portfolio_percentage * btc_holdings
            # btc_holdings = btc_holdings * (1 - portfolio_percentage)
            
            # tính theo số lượng BTC
            btc_holdings -= rdca_buy_sell_btc
            bias_amount = -current_btc_price
        # Hold thì mọi thứ vẫn giữ lại không thay đổi về lượng BTC đang giữ
        total_cash += btc_holdings * (prices[i] - current_btc_price)
        current_btc_price = prices[i]
        remaining_cash -= bias_amount
        valuation.append(total_cash)
        result_array[i].append(btc_holdings) # Cho mục đích debug
        result_array[i].append(total_cash)
        result_array[i].append(remaining_cash)
        # print(f"result_array[{i}]: {result_array[i]}")

    # Tính toán Valuation cho phương pháp DCA
    def calculate_dca_valuation(dates, prices, initial_cash=1000000, daily_investment=dca_daily_investment):
        remaining_cash_dca = initial_cash
        btc_holdings_dca = 0
        valuation_dca = []
        result_array_dca = [] # Cho mục đích debug
        current_valuation_dca = initial_cash

        for i in range(len(dates)):
            # result_array_dca.append([current_valuation_dca, remaining_cash_dca])   
            # print(f"result_array_dca[{i}]: {result_array_dca[i]}") 
            valuation_dca.append(current_valuation_dca)
            # Mỗi ngày đầu tư một số tiền cố định vào Bitcoin
            btc_bought_dca = daily_investment / prices[i]
            btc_holdings_dca += btc_bought_dca
            remaining_cash_dca -= daily_investment
            # Tính toán tổng valuation DCA (tiền mặt + số lượng BTC đang nắm giữ theo giá hiện tại)
            current_valuation_dca = remaining_cash_dca + btc_holdings_dca * price 
                            
        return valuation_dca

    valuation_dca = calculate_dca_valuation(dates, prices)

    # Tính toán Valuation cho phương pháp all in BTC
    def calculate_all_in_BTC(dates, prices, initial_cash=1000000):
        current_btc_price = prices[0]
        current_valuation = initial_cash
        btc_holdings_dca = current_valuation / current_btc_price
        valuation = []
        daily_bias = 0
        for i in range(len(dates)):         
            valuation.append(current_valuation)
            # print(f"valuation[{i}]: {valuation[i]}")
            daily_bias = btc_holdings_dca * (prices[i] - current_btc_price)
            current_valuation += daily_bias
            current_btc_price = prices[i] 
                            
        return valuation

    valuation_all_in_BTC = calculate_all_in_BTC(dates, prices)
    # Tính toán valuation cho phương pháp DCA (đường màu tím)

    # Vẽ biểu đồ với valuation
    fig, ax1 = plt.subplots(figsize=(12, 6))

    # Vẽ biểu đồ target trên trục thứ nhất
    # ax1.plot(dates, targets, label="Target (Buy=1, Hold=0, Sell=2)", marker='o', linestyle='-', color='black')

    # Cài đặt cho trục thứ nhất
    # ax1.set_xlabel("Date")
    # ax1.set_ylabel("Values")
    ax1.set_ylabel("BTC")
    ax1.set_xticks(dates)
    ax1.set_xticklabels(dates.strftime('%Y-%m-%d'), rotation=1)
    # ax1.set_xticklabels(dates.strftime('%m-%d'), rotation=days_observed)
    ax1.xaxis.set_major_locator(mdates.DayLocator(interval=days_interval))  # Hiển thị interval ngày
    # ax1.legend(loc="upper left")
    # ax1.grid(True)

    # Vẽ biểu đồ giá Bitcoin trên trục thứ hai
    # ax2 = ax1.twinx()
    ax1.scatter(dates, prices, label="BTC Price", color='orange', marker='o')
    # ax1.set_ylabel("BTC")
    ax1.legend(loc="upper right")

    # Vẽ biểu đồ valuation (giá trị tài sản)
    ax3 = ax1.twinx()
    # ax3.spines['right'].set_position(('outward', days_observed))  # Di chuyển trục thứ ba ra ngoài một chút
    ax3.plot(dates, valuation, label=f"R-DCA Buy or Sell {rdca_buy_sell_btc} BTC", color='green', linestyle='-', marker='o')
    ax3.set_ylabel("Valuation")

    # Vẽ biểu đồ valuation (DCA - Đường màu tím)
    ax3.plot(dates, valuation_dca, label="DCA 20k daily", color='purple', linestyle='-', marker='o')
    ax3.plot(dates, valuation_all_in_BTC, label="ALL IN BTC", color='blue', linestyle='-', marker='o')

    ax1.yaxis.set_major_formatter(FuncFormatter(y_format))
    ax3.legend(loc="upper left")

    # Thêm tiêu đề và hiển thị biểu đồ
    plt.title(f"Comparasion from {from_date_str} to {to_date_str}")
    plt.tight_layout()
    plt.show()

    print(f"Tổng giá trị tài sản (R-DCA): {valuation[-1]:.2f} USD")
    print(f"Tổng giá trị tài sản (DCA): {valuation_dca[-1]:.2f} USD")
    print(f"Tổng giá trị tài sản ALL IN BTC: {valuation_all_in_BTC[-1]:.2f} USD")
