def render_charts_date_range(from_date_str, to_date_str, halving=3, days_interval = 200):
    rdca_buy_sell_btc = constants.RDCA_BUY_SELL_BTC
    initial_cash = constants.INITIAL_CASH
    btc_initial = constants.BTC_INITIAL
    daily_investment = constants.DCA_BUY_SELL_USD
    trained_from, trained_to, _, _ = date_range_for_halving(halving)
    # Load mô hình đã lưu
    model = load_model(f"R_DCA_Halving_{halving}.keras")
    X_test_scaled, test_data = preprocess_data_for_chart(data, trained_from, trained_to, from_date_str, to_date_str)
    # print(f"test_data {test_data[-10:]}")

    # Dự đoán kết quả
    predictions = model.predict(X_test_scaled, verbose=0)
    # predicted_target = np.argmax(predictions[0], axis=1)

    # Áp dụng ngưỡng thay vì dùng argmax
    predicted_target = []
    threshold = 0.1  # Ví dụ: ngưỡng đang muốn là 10%
    for i, pred in enumerate(predictions[0]):
        target = 0
        if pred[1] > threshold:  # Lớp Buy có xác suất > ngưỡng
            target = 1
        elif pred[2] > threshold:  # Lớp Sell có xác suất > ngưỡng
            target = 2
        else:
            target = 0
        predicted_target.append(target)  # Hold nếu không có lớp nào vượt ngưỡng
        # print(f"Row {i} - Hold: {pred[0]:.4f}, Buy: {pred[1]:.4f}, Sell: {pred[2]:.4f} Target {target}")
    from_date = pd.to_datetime(from_date_str)  # Chuyển đổi thành Timestamp
    to_date = pd.to_datetime(to_date_str)    # Chuyển đổi thành Timestamp

    # Chuyển đổi cột 'Date' trong test_data thành kiểu datetime
    test_data['Date'] = pd.to_datetime(test_data['Date'])
    # print(f"------- test_data['Date'] {test_data['Date']}")

    # Lọc dữ liệu theo khoảng thời gian from_date đến to_date
    filtered_data = test_data[(test_data['Date'] >= from_date) & (test_data['Date'] <= to_date)]

    # Chuẩn bị dữ liệu để vẽ biểu đồ
    dates = filtered_data['Date'].dt.strftime('%Y-%m-%d').values  # Ngày trong khoảng thời gian từ from_date đến to_date, định dạng chuỗi 'YYYY-MM-DD'
    prices = filtered_data['Price'].values  # Giá Bitcoin trong khoảng thời gian đó

    # Tạo mảng kết quả theo ngày, target và confidence level
    result_array = []

    for i in range(len(test_data['Date'])):
        date = test_data['Date'].iloc[i]  # Lấy ngày từ dữ liệu test
        target = target_labels[predicted_target[i]]
        # Chỉ thêm kết quả nếu ngày nằm trong khoảng from_date đến to_date
        if from_date <= date <= to_date:
            # result_array.append([date.strftime('%Y-%m-%d'), target, price])  # Chuyển đổi 'date' sang định dạng chuỗi 'YYYY-MM-DD'
            result_array.append([date.strftime('%Y-%m-%d'), target])

    # # Danh sách ngày, target, giá Bitcoin từ dữ liệu
    dates = pd.to_datetime([row[0] for row in result_array])  # Convert list to pandas datetime

    # Tính toán Valuation cho phương pháp all in BTC
    ratio_profit_data = filtered_data['Ratio_Profit'].values
    ratio_mvrv_data = filtered_data['Ratio_MRRV'].values
    ratio_osci_data = filtered_data['Oscillator'].values
    risk_factors = filtered_data['Risk_Factor'].values  # Risk Factor để tính toán ra Buy Amount

    # Cập nhật valuation theo target và giá Bitcoin
    targets = [target_mapping[row[1]] for row in result_array]

    # Vẽ thêm option của profit nhưng buy/sell theo USD thay vì BTC
    valuation_by_profit_usd, btc_holdings_by_profit_usd, data_table_profit_usd, buy_number_profit_usd, sell_number_profit_usd = calculate_valuation_by_profit_USD(dates, prices, ratio_profit_data, risk_factors)
    # display_data_with_columns(data_table_profit_usd, ["Date", "BTC", "Profit", "Valuation", "BTC holding", "Remaining Cash"])
    
    valuation_by_profit, btc_holdings_by_profit, data_table_profit, buy_number_profit, sell_number_profit = calculate_valuation_by_profit_BTC(dates, prices, ratio_profit_data, risk_factors)
    # display_data_with_columns(data_table_profit, ["Date", "BTC", "Profit", "Valuation", "BTC holding", "Remaining Cash"])

    # valuation_by_mvrv, btc_holdings_by_mvrv, data_table_mvrv, buy_number_mvrv, sell_number_mvrv = calculate_valuation_by_mvrv(dates, prices, ratio_mvrv_data)
    # display_data_with_columns(data_table_mvrv, ["Date", "BTC", "MVRV", "Valuation", "BTC holding", "Remaining Cash"])

    # valuation_by_osci, btc_holdings_by_osci, data_table_osci, buy_number_osci, sell_number_osci = calculate_valuation_by_oscillator(dates, prices, ratio_osci_data)
    # display_data_with_columns(data_table_osci, ["Date", "BTC", "Osci", "Valuation", "BTC holding", "Remaining Cash"])

    valuation_by_ALL_IN, btc_holdings_by_ALL_IN, data_table_all_in, buy_number_all_in = calculate_valuation_by_ALL_IN(dates, prices)
    # display_data_with_columns(data_table_all_in, ["Date", "BTC", "Valuation", "BTC holding", "Remaining Cash"])

    valuation_by_DCA, btc_holdings_by_DCA, data_table_dca, buy_number_dca = calculate_valuation_by_DCA(dates, prices)

    # Tính Drawdown cho mỗi chiến lược
    def calculate_drawdown(valuation_series):
        valuation_series = pd.Series(valuation_series)  # Chuyển đổi list thành pandas Series
        peak = valuation_series.cummax()  # Tính giá trị cao nhất tại mỗi thời điểm
        drawdown = (valuation_series - peak) / peak  # Tính Drawdown dưới dạng phần trăm
        return drawdown

def display_data_with_columns(data, columns_display=["Date", "BTC", "Metric", "Valuation", "BTC holding", "Remaining Cash"]):
    row_display_limit = 0
    from IPython.display import display
    df = pd.DataFrame(data, columns=columns_display)
    # Ví dụ với DataFrame `df`
    df.style.set_table_styles([
        {'selector': 'th', 'props': [('background-color', '#f0f0f0'), ('font-weight', 'bold'), ('text-align', 'center')]},
        {'selector': 'td', 'props': [('text-align', 'center')]}
    ])
    # Thiết lập định dạng số để hiển thị đầy đủ
    pd.options.display.float_format = '{:,.2f}'.format  # Số thập phân có thể điều chỉnh với .2f hoặc .0f để bỏ thập phân
    pd.set_option('display.max_rows', None)
    limited_df = df
    # Limit với số dòng nhất định
    if row_display_limit > 0:
        head_df = df.head(row_display_limit)
        tail_df = df.tail(row_display_limit)
        limited_df = pd.concat([head_df, tail_df])
    elif row_display_limit <= 0:
        return
    display(limited_df)

def print_all_constants():
    # Get all attributes of the constants module
    all_attributes = dir(constants)

    output = "Constants defined in constants.py:\n\n"
    for attr_name in all_attributes:
        # Skip built-in attributes and methods
        if not attr_name.startswith("__"):
            attr_value = getattr(constants, attr_name)
            # Check if it's a simple type (assuming constants are simple types)
            if isinstance(attr_value, (int, float, str, bool)):
                output += f"**{attr_name}**: {attr_value}\n\n"

    display(Markdown(output))
