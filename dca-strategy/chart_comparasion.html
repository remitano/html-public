import numpy as np

# Load m hnh  lu
model = load_model("R_DCA_T2_trained.keras")

from matplotlib.ticker import FuncFormatter

# Hm formatter  hin th gi tr Bitcoin ngn gn
def y_format(value, tick_number):
    return f'{int(value/1000)}k'  # Chuyn gi tr thnh nh dng '68k'
# Load v sort li data theo ngy tng dn  render chart
data = pd.read_csv("../../data/merged_data_train_test_output.csv").sort_values(by='Date', ascending=True)

def render_chart_with_param(from_date_str, to_date_str, days_interval = 7, rdca_buy_sell_btc = 5, total_cash=1000000):
    from_date = pd.to_datetime(from_date_str)  # Chuyn i thnh Timestamp
    to_date = pd.to_datetime(to_date_str)    # Chuyn i thnh Timestamp

    # Chuyn i ct 'Date' trong test_data thnh kiu datetime
    test_data['Date'] = pd.to_datetime(test_data['Date'])

    # Lc d liu theo khong thi gian from_date n to_date
    filtered_data = test_data[(test_data['Date'] >= from_date) & (test_data['Date'] <= to_date)]

    # Chun b d liu  v biu 
    dates = filtered_data['Date'].dt.strftime('%Y-%m-%d').values  # Ngy trong khong thi gian t from_date n to_date, nh dng chui 'YYYY-MM-DD'
    prices = filtered_data['Price'].values  # Gi Bitcoin trong khong thi gian 

    # To mng kt qu theo ngy, target v confidence level
    result_array = []

    for i in range(len(predicted_target)):
        date = test_data['Date'].iloc[i]  # Ly ngy t d liu test
        # print(f"PREDICTIONS {i} predicted_target[i] {predicted_target[i]}")
        target = target_labels[predicted_target[i]]  # Chuyn target sang chui Buy/Hold/Sell
        price = test_data['Price'].iloc[i]
        
        # Ch thm kt qu nu ngy nm trong khong from_date n to_date
        if from_date <= date <= to_date:
            result_array.append([date.strftime('%Y-%m-%d'), target, price])  # Chuyn i 'date' sang nh dng chui 'YYYY-MM-DD'
    # In kt qu result_array  kim tra
    # print(f"Filtered result_array from {from_date} to {to_date}:")
    # for row in result_array:
    #     print(row)

    # raise SystemExit
    # Danh sch ngy, target, gi Bitcoin t d liu
    dates = pd.to_datetime([row[0] for row in result_array])  # Convert list to pandas datetime
    targets = [target_mapping[row[1]] for row in result_array]

    current_btc_price = prices[0]

    remaining_cash = total_cash * (1 - portfolio_percentage)
    btc_holdings = (total_cash *  portfolio_percentage) / current_btc_price  # Bt u vi 1 triu USD, u t 30% vo Bitcoin
    valuation = []  # Danh sch cha gi tr ti sn mi ngy (tin mt + gi tr Bitcoin)


    # Cp nht valuation theo target v gi Bitcoin
    for i in range(len(dates)):
        action = targets[i]  # 0 = Hold, 1 = Buy, 2 = Sell
        bias_amount = 0
        if action == 1:  # Buy Bitcoin
            # Tnh theo %
            # bias_amount = remaining_cash * portfolio_percentage
            # btc_holdings += bias_amount / current_btc_price
            
            # tnh theo s lng BTC
            btc_holdings += rdca_buy_sell_btc
            bias_amount = current_btc_price
        elif action == 2:  # Sell Bitcoin
            # Tnh theo %
            # bias_amount = - prices[i] * portfolio_percentage * btc_holdings
            # btc_holdings = btc_holdings * (1 - portfolio_percentage)
            
            # tnh theo s lng BTC
            btc_holdings -= rdca_buy_sell_btc
            bias_amount = -current_btc_price
        # Hold th mi th vn gi li khng thay i v lng BTC ang gi
        total_cash += btc_holdings * (prices[i] - current_btc_price)
        current_btc_price = prices[i]
        remaining_cash -= bias_amount
        valuation.append(total_cash)
        result_array[i].append(btc_holdings) # Cho mc ch debug
        result_array[i].append(total_cash)
        result_array[i].append(remaining_cash)
        # print(f"result_array[{i}]: {result_array[i]}")

    # Tnh ton Valuation cho phng php DCA
    def calculate_dca_valuation(dates, prices, initial_cash=1000000, daily_investment=dca_daily_investment):
        remaining_cash_dca = initial_cash
        btc_holdings_dca = 0
        valuation_dca = []
        result_array_dca = [] # Cho mc ch debug
        current_valuation_dca = initial_cash

        for i in range(len(dates)):
            # result_array_dca.append([current_valuation_dca, remaining_cash_dca])   
            # print(f"result_array_dca[{i}]: {result_array_dca[i]}") 
            valuation_dca.append(current_valuation_dca)
            # Mi ngy u t mt s tin c nh vo Bitcoin
            btc_bought_dca = daily_investment / prices[i]
            btc_holdings_dca += btc_bought_dca
            remaining_cash_dca -= daily_investment
            # Tnh ton tng valuation DCA (tin mt + s lng BTC ang nm gi theo gi hin ti)
            current_valuation_dca = remaining_cash_dca + btc_holdings_dca * price 
                            
        return valuation_dca

    valuation_dca = calculate_dca_valuation(dates, prices)

    # Tnh ton Valuation cho phng php all in BTC
    def calculate_all_in_BTC(dates, prices, initial_cash=1000000):
        current_btc_price = prices[0]
        current_valuation = initial_cash
        btc_holdings_dca = current_valuation / current_btc_price
        valuation = []
        daily_bias = 0
        for i in range(len(dates)):         
            valuation.append(current_valuation)
            # print(f"valuation[{i}]: {valuation[i]}")
            daily_bias = btc_holdings_dca * (prices[i] - current_btc_price)
            current_valuation += daily_bias
            current_btc_price = prices[i] 
                            
        return valuation

    valuation_all_in_BTC = calculate_all_in_BTC(dates, prices)
    # Tnh ton valuation cho phng php DCA (ng mu tm)

    # V biu  vi valuation
    fig, ax1 = plt.subplots(figsize=(12, 6))

    # V biu  target trn trc th nht
    # ax1.plot(dates, targets, label="Target (Buy=1, Hold=0, Sell=2)", marker='o', linestyle='-', color='black')

    # Ci t cho trc th nht
    # ax1.set_xlabel("Date")
    # ax1.set_ylabel("Values")
    ax1.set_ylabel("BTC")
    ax1.set_xticks(dates)
    ax1.set_xticklabels(dates.strftime('%Y-%m-%d'), rotation=1)
    # ax1.set_xticklabels(dates.strftime('%m-%d'), rotation=days_observed)
    ax1.xaxis.set_major_locator(mdates.DayLocator(interval=days_interval))  # Hin th interval ngy
    # ax1.legend(loc="upper left")
    # ax1.grid(True)

    # V biu  gi Bitcoin trn trc th hai
    # ax2 = ax1.twinx()
    ax1.scatter(dates, prices, label="BTC Price", color='orange', marker='o')
    # ax1.set_ylabel("BTC")
    ax1.legend(loc="upper right")

    # V biu  valuation (gi tr ti sn)
    ax3 = ax1.twinx()
    # ax3.spines['right'].set_position(('outward', days_observed))  # Di chuyn trc th ba ra ngoi mt cht
    ax3.plot(dates, valuation, label=f"R-DCA Buy or Sell {rdca_buy_sell_btc} BTC", color='green', linestyle='-', marker='o')
    ax3.set_ylabel("Valuation")

    # V biu  valuation (DCA - ng mu tm)
    ax3.plot(dates, valuation_dca, label="DCA 20k daily", color='purple', linestyle='-', marker='o')
    ax3.plot(dates, valuation_all_in_BTC, label="ALL IN BTC", color='blue', linestyle='-', marker='o')

    ax1.yaxis.set_major_formatter(FuncFormatter(y_format))
    ax3.legend(loc="upper left")

    # Thm tiu  v hin th biu 
    plt.title(f"Comparasion from {from_date_str} to {to_date_str}")
    plt.tight_layout()
    plt.show()

    print(f"Tng gi tr ti sn (R-DCA): {valuation[-1]:.2f} USD")
    print(f"Tng gi tr ti sn (DCA): {valuation_dca[-1]:.2f} USD")
    print(f"Tng gi tr ti sn ALL IN BTC: {valuation_all_in_BTC[-1]:.2f} USD")
